async function exampleWithAsyncAwait() {

Q7 Create examples to explain promises function


function asyncOperation() {
                    return new Promise((resolve, reject) => {
                      setTimeout(() => {
                        const randomValue = Math.random();
                        if (randomValue >= 0.5) {
                          resolve(randomValue);
                        } else {
                          reject(new Error('Random value is less than 0.5'));
                        }
                      }, 1000);
                    });
                  }
                  
                  asyncOperation()
                    .then((result) => console.log('Resolved:', result))
                    .catch((error) => console.error('Rejected:', error.message));

Ans
Rejected: Random value is less than 0.5

Q9 Create examples to explain promise.all function


function asyncOperation1() {
                            return new Promise((resolve) => {
                              setTimeout(() => {
                                resolve('Result from asyncOperation1');
                              }, 2000);
                            });
                          }
                          
                          function asyncOperation2() {
                            return new Promise((resolve) => {
                              setTimeout(() => {
                                resolve('Result from asyncOperation2');
                              }, 1500);
                            });
                          }
                          async function executeAsyncOperations() {
                            try {
                              const results = await Promise.all([asyncOperation1(), asyncOperation2()]);
                              console.log('All results:', results);
                            } catch (error) {
                              console.error('Error occurred:', error.message);
                            }
                          }
                          
                          executeAsyncOperations();

Ans
All results: [ 'Result from asyncOperation1', 'Result from asyncOperation2' ]
